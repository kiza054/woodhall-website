class ContactUsView(generic.View):
    def get(self, request):
        articles = Article.objects.filter(status=1).order_by('-date_posted')[:2]
        form = ContactForm
        context = {
            'title': 'Contact Us',
            'form': form,
            'articles': articles
        }
        return render(request, 'main_website/contact_us.html', context)
    def post(self, request):
        form = ContactForm(request.POST)
        articles = Article.objects.filter(status=1).order_by('-date_posted')[:2]
        context = {
            'title': 'Contact Us',
            'form': form,
            'articles': articles
        }
        if form.is_valid():
            subject = form.cleaned_data['subject']
            contact_email = form.cleaned_data['contact_email']
            content = form.cleaned_data['content']
            try:
                send_mail(subject, content, contact_email, ['k.sarda.scouts@gmail.com'])
            except BadHeaderError:
                return HttpResponse('Invalid header found')
            messages.success(request, f'Your message has been sent!')
        return render(request, 'main_website/contact_us.html', context)
